  470.      *  HTML filter first. If set, no more filtering is performed. If no bits
  471.      *  other than the 1 bit is set, a strict filter is applied.

  475.     {

  496.             }

  499.         else

  502.             if (is_null($noHtmlFilter)) {
  503.                 $noHtmlFilter = & JFilterInput::getInstance(/* $tags, $attr, $tag_method, $attr_method, $xss_auto */);
